# tampilan-termux
Script ini untuk membuat tampilan Termux mu lebih keren:v

# Cara Menginstall Script
# 👇🏻


$ pkg update && pkg upgrade

$ pkg up -y

$ pkg in git python2 -y

$ git clone https://github.com/reytnk/tampilan-termux

$ cd tampilan-termux

$ python2 tampilan.py
